# I2C address of the PCF8591 (usually 0x48) ADDRESS = 0x48

# Initialize I2C bus bus = smbus.SMBus(1)

def read_analog(channel): """ Reads analog value from PCF8591. Channel range: 0 to 3 (AIN0 to AIN3) """ # 0x40 is the control byte for AIN0 bus.write_byte(ADDRESS, 0x40 + channel)

# Dummy read to start conversion bus.read_byte(ADDRESS)

# Actual read (returns 0-255 for 8-bit precision) return bus.read_byte(ADDRESS)

try: print("Reading Hall sensor analog values... Press Ctrl+C to stop.") while True: value = read_analog(0) # Reading from AIN0 voltage = value * (3.3 / 255.0) # Convert 0-255 to 0-3.3V

print(f"Raw Value: {value} | Voltage: {voltage:.2f}V") time.sleep(0.5)

except KeyboardInterrupt: print("\nProgram stopped.")
